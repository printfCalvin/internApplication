#include <iostream>
#include <cassert>
using namespace std;

struct Queue {
  int size;
  int frt;
  int last;
  int *arr;
};

void init(Queue &q, int size) {
  q.size = size;
  q.frt = -1;
  q.last = -1;
  q.arr = new int [size];
}

void add(Queue &q, int val) {
  if(q.frt == -1) {
    assert(q.last == -1);
    q.frt = 0;
    q.last = 0;
    return;
  }

  if(q.last == q.size - 1) {
    q.last = 0;
  } else {
    q.last++;
  }
  q.arr[q.last] = val;
  if(q.frt == q.last) {
    q.frt++;
  }
}

void remove(Queue &q) {
  assert(q.frt != -1 && q.last != -1);
  if(q.last == q.frt) {
    q.last = -1;
    q.frt = -1;
  } else if(q.frt + 1 == q.size) {
    q.frt = 0;
  } else {
    q.frt++;
  }
}

void print(Queue &q) {
  cout << "frt is " << q.frt << " last is " << q.last << endl;
  int i = q.frt;
  while(true) {
    cout << q.arr[i] << ' ';
    if(i == q.last) break;
    else if(i + 1 == q.size) i = 0;
    else i++;
  }
  cout << endl;
}
